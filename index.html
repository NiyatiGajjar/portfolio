oil_well_pipeline.py — IoT Serial Data Ingestion & Motor Alert System
# Oil Well Depth Monitoring — Niyati Gajjar · QuantumBot Pvt. Ltd.
# Serial data from STM32 → noise filter → Django dashboard → motor trigger

import serial, requests, json, time, logging
from collections import deque
from datetime import datetime

SERIAL_PORT     = "/dev/ttyUSB0"
BAUD_RATE       = 115200
SAMPLE_INTERVAL = 0.05    # 50ms — millisecond sampling
DEPTH_THRESHOLD = 15.0   # meters — motor triggers below this
DJANGO_API      = "http://dashboard.local/api/well/readings/"

buffer = deque(maxlen=20)  # rolling average for noise filtering

def filter_noise(raw):
    buffer.append(raw)
    return sum(buffer) / len(buffer)

def parse_frame(raw_bytes):
    try:
        p = raw_bytes.decode("utf-8").strip().split("|")
        if p[0] != "START": return None
        return {"depth": float(p[1]), "temp": float(p[2]), "pressure": float(p[3])}
    except: return None

def send_to_dashboard(depth, temp, pressure, alert):
    try:
        requests.post(DJANGO_API, json={
            "depth_m": round(depth, 3), "temp": temp,
            "alert": alert, "ts": datetime.utcnow().isoformat()
        }, timeout=3)
        if alert: logging.warning(f"⚠ Depth {depth:.2f}m — Motor triggered")
    except Exception as e: logging.error(f"API: {e}")

def main():
    with serial.Serial(SERIAL_PORT, BAUD_RATE, timeout=1) as ser:
        while True:
            data = parse_frame(ser.readline())
            if not data: continue
            d = filter_noise(data["depth"])
            send_to_dashboard(d, data["temp"], data["pressure"], d < DEPTH_THRESHOLD)
            time.sleep(SAMPLE_INTERVAL)

if __name__ == "__main__": main()
