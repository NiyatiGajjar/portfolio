oil_well_pipeline.py — IoT Serial Data Ingestion & Motor Alert System 
# Oil Well Depth Monitoring — Serial Data Pipeline
# Niyati Gajjar · QuantumBot Pvt. Ltd.
# Reads STM32 sensor data via serial port (50ms intervals),
# filters noise with rolling average, computes depth,
# sends to Django dashboard via REST API, triggers motor alert.

import serial, requests, json, time, logging
from collections import deque
from datetime import datetime

# ── Configuration ──────────────────────────────────
SERIAL_PORT     = "/dev/ttyUSB0"
BAUD_RATE       = 115200
SAMPLE_INTERVAL = 0.05    # 50ms — millisecond-level sampling
NOISE_WINDOW    = 20      # rolling average window size
DEPTH_THRESHOLD = 15.0   # meters — trigger motor below this
DJANGO_API      = "http://dashboard.local/api/well/readings/"
MOTOR_API       = "http://dashboard.local/api/well/motor/trigger/"
WELL_ID         = "WELL_001"

log = logging.getLogger(__name__)
logging.basicConfig(level=logging.INFO,
    format="%(asctime)s [%(levelname)s] %(message)s")

# ── Noise Filter — Rolling Average ─────────────────
buffer = deque(maxlen=NOISE_WINDOW)

def filter_noise(raw: float) -> float:
    """Smooth sensor readings with rolling average."""
    buffer.append(raw)
    return sum(buffer) / len(buffer)

# ── Parse STM32 Serial Frame ────────────────────────
def parse_frame(raw_bytes: bytes) -> dict | None:
    """Frame format: START|DEPTH|TEMP|PRESSURE|END"""
    try:
        frame = raw_bytes.decode("utf-8").strip()
        if not frame.startswith("START"): return None
        p = frame.split("|")
        return {"depth": float(p[1]), "temp": float(p[2]), "pressure": float(p[3])}
    except (ValueError, IndexError): return None

# ── Push to Django Dashboard ────────────────────────
def send_to_dashboard(depth, temp, pressure, alert):
    payload = {"well_id": WELL_ID, "depth_m": round(depth, 3),
               "temperature": temp, "pressure": pressure,
               "alert": alert, "timestamp": datetime.utcnow().isoformat()}
    hdrs = {"Content-Type": "application/json"}
    try:
        requests.post(DJANGO_API, data=json.dumps(payload), headers=hdrs, timeout=3)
        if alert:
            requests.post(MOTOR_API, headers=hdrs, timeout=3,
                data=json.dumps({"well_id": WELL_ID, "action": "START_FILL"}))
            log.warning(f"ALERT: depth {depth:.2f}m below threshold. Motor triggered.")
    except requests.RequestException as e:
        log.error(f"API error: {e}")

# ── Main Loop ───────────────────────────────────────
def main():
    log.info(f"Connecting to STM32 on {SERIAL_PORT}…")
    with serial.Serial(SERIAL_PORT, BAUD_RATE, timeout=1) as ser:
        log.info("Connected. Reading sensor data…")
        while True:
            raw = ser.readline()
            if not raw: continue
            data = parse_frame(raw)
            if data is None: continue
            depth_smooth = filter_noise(data["depth"])
            send_to_dashboard(depth_smooth, data["temp"],
                              data["pressure"], depth_smooth < DEPTH_THRESHOLD)
            time.sleep(SAMPLE_INTERVAL)

if __name__ == "__main__": main()
